<script>
document.addEventListener('DOMContentLoaded', () => {
    const vessel = document.getElementById('vessel');
    const partition = document.getElementById('partition');
    const particlesContainer = document.getElementById('particles');
    const removePartitionButton = document.getElementById('removePartitionButton');
    const restartSimulationButton = document.getElementById('restartSimulationButton');
    const toggleExplanationButton = document.getElementById('toggleExplanationButton');
    const explanationDiv = document.getElementById('explanation');
    const leftCountSpan = document.getElementById('leftCount');
    const rightCountSpan = document.getElementById('rightCount');
    const statusTextSpan = document.getElementById('statusText');

    const numParticles = 200; // ×”×’×“×œ×ª ××¡×¤×¨ ×”×—×œ×§×™×§×™× ×œ××¤×§×˜ ×•×™×–×•××œ×™ ×˜×•×‘ ×™×•×ª×¨
    const particleSize = 4; // ×’×•×“×œ ×—×œ×§×™×§ ××¢×˜ ×§×˜×Ÿ ×™×•×ª×¨
    const maxSpeed = 2; // ××”×™×¨×•×ª ××™×¨×‘×™×ª ×’×‘×•×”×” ×™×•×ª×¨
    const boundaryMargin = particleSize / 2; // ×©×•×œ×™×™× ×§×˜× ×™× ××’×‘×•×œ ×”×›×œ×™

    let particles = [];
    let animationFrameId;
    let partitionRemoved = false;
    let vesselWidth = vessel.clientWidth; // ×™×§×‘×¢ ×‘×”×ª×—×œ×” ×•×‘×¢×ª ×©×™× ×•×™ ×’×•×“×œ
    let vesselHeight = vessel.clientHeight;
    let initialPartitionX = vesselWidth / 2;

    // ×¤×•× ×§×¦×™×” ×œ×¢×“×›×•×Ÿ ××™××“×™ ×”×›×œ×™ (×œ××§×¨×” ×©×œ ×©×™× ×•×™ ×’×•×“×œ ×—×œ×•×Ÿ)
    function updateVesselDimensions() {
         vesselWidth = vessel.clientWidth;
         vesselHeight = vessel.clientHeight;
         initialPartitionX = vesselWidth / 2;
          // ×¢×“×›×Ÿ ×’×‘×•×œ×•×ª ×”×ª× ×’×©×•×ª ×‘××™×“×ª ×”×¦×•×¨×š (×œ× ×”×›×¨×—×™ ×›×¨×’×¢ ×›×™ ×”× ×“×™× ××™×™×)
    }

    // ×¤×•× ×§×¦×™×” ×œ×™×¦×™×¨×ª ×—×œ×§×™×§ ×‘×•×“×“
    function createParticle(x, y) {
        const particle = document.createElement('div');
        particle.classList.add('particle');
        particle.style.width = `${particleSize}px`;
        particle.style.height = `${particleSize}px`;
        particlesContainer.appendChild(particle);

        // ××”×™×¨×•×ª ×”×ª×—×œ×ª×™×ª ××§×¨××™×ª
        const angle = Math.random() * 2 * Math.PI;
        const speed = Math.random() * maxSpeed * 0.5 + maxSpeed * 0.5; // ××”×™×¨×•×ª ×‘×™×Ÿ ××—×¦×™×ª ×”××”×™×¨×•×ª ×”××™×¨×‘×™×ª ×œ××™×¨×‘×™×ª ×¢×¦××”

        return {
            element: particle,
            x: x,
            y: y,
            vx: Math.cos(angle) * speed,
            vy: Math.sin(angle) * speed
        };
    }

    // ××ª×—×•×œ ×—×œ×§×™×§×™× ×‘×¦×“ ×©×××œ ×‘×œ×‘×“
    function initializeParticles() {
        // ×¢×¦×•×¨ ×× ×™××¦×™×” ×× ×¨×¦×”
        if (animationFrameId) {
            cancelAnimationFrame(animationFrameId);
        }

        particlesContainer.innerHTML = ''; // × ×§×” ×—×œ×§×™×§×™× ×§×™×™××™×
        particles = [];
        partitionRemoved = false;

        // ×•×“× ×©×”××—×™×¦×” ×’×œ×•×™×” ×•×‘××§×•×
        partition.style.display = 'block';
        partition.classList.remove('removed'); // ×”×¡×¨ ×§×œ××¡ ×× ×™××¦×™×” ×× ×”×™×”
        partition.style.left = '50%'; // ×•×“× ×©×”××—×™×¦×” ×‘×“×™×•×§ ×‘×××¦×¢

        // ×•×“× ×©×”×›×¤×ª×•×¨ ×¤×¢×™×œ ×•×¢× ×”×˜×§×¡×˜ ×”× ×›×•×Ÿ
        removePartitionButton.disabled = false;
        removePartitionButton.textContent = "×”×¡×¨ ××ª ×”××—×™×¦×” ×•×¦×¤×” ××” ×§×•×¨×”!";
        removePartitionButton.style.backgroundColor = ''; // ×”×¡×¨ ×¦×‘×¢ ××¤×•×¨

        // ××ª×—×œ ××•× ×™ ×—×œ×§×™×§×™×
        updateCounts();


        updateVesselDimensions(); // ×¢×“×›×Ÿ ××™××“×™× ×œ×¤× ×™ ××ª×—×•×œ ××™×§×•× ×—×œ×§×™×§×™×

        for (let i = 0; i < numParticles; i++) {
            // ××§× ×‘××§×¨××™ ×‘×—×¦×™ ×”×©×××œ×™ ×©×œ ×”×›×œ×™, ×¢× ×©×•×œ×™×™× ×§×˜× ×™× ××”×§×™×¨×•×ª ×•×”××—×™×¦×”
            const x = Math.random() * (initialPartitionX - particleSize * 2 - boundaryMargin * 2) + boundaryMargin;
            const y = Math.random() * (vesselHeight - particleSize * 2 - boundaryMargin * 2) + boundaryMargin;
            particles.push(createParticle(x, y));
        }
        renderParticles(); // ×”×¦×’×” ×¨××©×•× ×™×ª
        animate(); // ×”×ª×—×œ ×× ×™××¦×™×” ××™×“ ×›×“×™ ×œ×”×¨××•×ª ××ª ×”×ª× ×•×¢×” ×‘×ª×•×š ×”×¦×“ ×”×©×××œ×™
    }

    // ×¨×™× ×“×•×¨ ××™×§×•××™ ×—×œ×§×™×§×™× ×¢×œ ×”××¡×š
    function renderParticles() {
        particles.forEach(p => {
            p.element.style.left = `${p.x}px`;
            p.element.style.top = `${p.y}px`;
        });
    }

    // ×¤×•× ×§×¦×™×” ×œ×¢×“×›×•×Ÿ ××•× ×™ ×—×œ×§×™×§×™× ×•×¡×˜×˜×•×¡
    function updateCounts() {
        let leftCount = 0;
        let rightCount = 0;

        particles.forEach(p => {
             // ×—×©×‘ ×œ×¤×™ ××¨×›×– ×”×—×œ×§×™×§ ×œ×¢×•××ª ××¨×›×– ×”×›×œ×™/××—×™×¦×”
             if (p.x + particleSize / 2 < vesselWidth / 2) {
                 leftCount++;
             } else {
                 rightCount++;
             }
        });

        leftCountSpan.textContent = `×—×œ×§×™×§×™× ×‘×©×××œ: ${leftCount}`;
        rightCountSpan.textContent = `×—×œ×§×™×§×™× ×‘×™××™×Ÿ: ${rightCount}`;

        // ×¢×“×›×Ÿ ×¡×˜×˜×•×¡ ×˜×§×¡×˜×•××œ×™
        if (!partitionRemoved) {
             statusTextSpan.textContent = ' | ';
        } else {
             // ×—×©×‘ ×™×—×¡
             const total = leftCount + rightCount;
             if (total === 0) {
                  statusTextSpan.textContent = ' | ';
                  return;
             }
             const leftRatio = leftCount / total;
             const rightRatio = rightCount / total;

             if (leftCount === total || rightCount === total) {
                  statusTextSpan.textContent = ' | ××¦×‘ ×”×ª×—×œ×ª×™/×××•×“ ××¡×•×“×¨ | ';
             } else if (leftRatio > 0.45 && leftRatio < 0.55) { // ×‘×¢×¨×š 50/50
                  statusTextSpan.textContent = ' | ××¦×‘ ×§×¨×•×‘ ×œ×©×™×•×•×™ ××©×§×œ (×‘×œ×’×Ÿ ××¨×‘×™) | ';
             } else if (leftRatio > 0.2 && leftRatio < 0.8) { // ×¤×™×–×•×¨ ×¡×‘×™×¨
                 statusTextSpan.textContent = ' | ××ª×¤×–×¨×™×... | ';
             } else { // ×¢×“×™×™×Ÿ ××¨×•×›×–×™× ×™×—×¡×™×ª
                 statusTextSpan.textContent = ' | ×¤×™×–×•×¨ ×”×ª×—×œ×ª×™... | ';
             }
        }
    }


    // ×œ×•×œ××ª ×× ×™××¦×™×”
    function animate() {
        updateVesselDimensions(); // ×•×“× ××™××“×™× ××¢×•×“×›× ×™× ×œ×—×™×©×•×‘×™ ×”×ª× ×’×©×•×ª

        particles.forEach(p => {
            // ×¢×“×›×•×Ÿ ××™×§×•×
            p.x += p.vx;
            p.y += p.vy;

            // ×˜×™×¤×•×œ ×‘×”×ª× ×’×©×•×™×•×ª ×¢× ×”×§×™×¨×•×ª (×”×§×¤×¦×”)
            // ×§×™×¨ ×©×××œ
            if (p.x < boundaryMargin) {
                p.x = boundaryMargin + (boundaryMargin - p.x); // ×ª×§×Ÿ ××™×§×•×
                p.vx *= -1; // ×”×¤×•×š ×›×™×•×•×Ÿ
            }
             // ×§×™×¨ ×™××™×Ÿ
            if (p.x > vesselWidth - particleSize - boundaryMargin) {

                p.vx *= -1; // ×”×¤×•×š ×›×™×•×•×Ÿ
            }
            // ×§×™×¨ ×¢×œ×™×•×Ÿ
            if (p.y < boundaryMargin) {
                p.y = boundaryMargin + (boundaryMargin - p.y); // ×ª×§×Ÿ ××™×§×•×
                p.vy *= -1; // ×”×¤×•×š ×›×™×•×•×Ÿ
            }
             // ×§×™×¨ ×ª×—×ª×•×Ÿ
            if (p.y > vesselHeight - particleSize - boundaryMargin) {
                p.y = vesselHeight - particleSize - boundaryMargin - (p.y - (vesselHeight - particleSize - boundaryMargin)); // ×ª×§×Ÿ ××™×§×•×
                p.vy *= -1; // ×”×¤×•×š ×›×™×•×•×Ÿ
            }


            // ×˜×™×¤×•×œ ×‘×”×ª× ×’×©×•×ª ×¢× ×”××—×™×¦×” ×× ×”×™× ×§×™×™××ª
            if (!partitionRemoved) {
                // × ×‘×“×•×§ ×”×ª× ×’×©×•×ª ×¨×§ ×× ×”×—×œ×§×™×§ ×§×¨×•×‘ ×œ××™×–×•×¨ ×”××—×™×¦×”
                 const partitionThreshold = initialPartitionX; // ×§×• ×”××—×™×¦×” ×”×××¦×¢×™
                 const particleCenter = p.x + particleSize / 2;
                 const prevParticleCenter = (p.x - p.vx) + particleSize / 2;


                // ×”×ª× ×’×©×•×ª ×›×©×”×—×œ×§×™×§ ×¢×•×‘×¨ ××¦×“ ×œ×¦×“ ××ª ×§×• ×”×××¦×¢
                if (
                    // ×–×– ×™××™× ×” ×•×—×•×¦×” ××ª ×§×• ×”×××¦×¢
                    p.vx > 0 && prevParticleCenter <= partitionThreshold && particleCenter > partitionThreshold
                    ||
                    // ×–×– ×©×××œ×” ×•×—×•×¦×” ××ª ×§×• ×”×××¦×¢
                    p.vx < 0 && prevParticleCenter >= partitionThreshold && particleCenter < partitionThreshold
                ) {
                     p.vx *= -1; // ×”×¤×•×š ××”×™×¨×•×ª ××•×¤×§×™×ª
                     // ×ª×§×Ÿ ××™×§×•× ×›×“×™ ×©×œ× ×™×ª×§×¢ ×‘××—×™×¦×” ××• ×™×¢×‘×•×¨ ××•×ª×”
                     if (p.vx > 0) { // ×”×—×œ×§×™×§ ×”×ª×§×“× ×©×××œ×” ×•×”×•×§×¤×¥ ×™××™× ×”
                          p.x = partitionThreshold + (partitionThreshold - particleCenter) - particleSize/2; // ××™×§×•× ××—×¨×™ ×”×”×§×¤×¦×”, ××™××™×Ÿ ×œ××—×™×¦×”
                     } else { // ×”×—×œ×§×™×§ ×”×ª×§×“× ×™××™× ×” ×•×”×•×§×¤×¥ ×©×××œ×”
                          p.x = partitionThreshold - (particleCenter - partitionThreshold) - particleSize/2; // ××™×§×•× ××—×¨×™ ×”×”×§×¤×¦×”, ××©×××œ ×œ××—×™×¦×”
                     }
                }
            }
        });

        renderParticles(); // ×¢×“×›×Ÿ DOM
        updateCounts(); // ×¢×“×›×Ÿ ××•× ×™ ×—×œ×§×™×§×™×
        animationFrameId = requestAnimationFrame(animate); // ×”××©×š ×× ×™××¦×™×”
    }

    // ××™×¨×•×¢ ×œ×—×™×¦×” ×¢×œ ×›×¤×ª×•×¨ ×”×¡×¨×ª ×”××—×™×¦×”
    removePartitionButton.addEventListener('click', () => {

            // ×”××—×™×¦×” ×ª×™×¢×œ× ×¤×™×–×™×ª ×œ××—×¨ ×”×× ×™××¦×™×” (××• ××™×™×“×™×ª, ×ª×œ×•×™ ×‘-CSS display)
            // ×›×“×™ ×©×”×”×ª× ×’×©×•×ª ×ª×¤×¡×™×§ ××™×“:
            partitionRemoved = true;
            removePartitionButton.disabled = true;
            removePartitionButton.textContent = "×”××—×™×¦×” ×”×•×¡×¨×”";
            removePartitionButton.style.backgroundColor = "#6c757d";
            removePartitionButton.style.cursor = "default";

            // ×× ×× ×™××¦×™×” ×¢×“×™×™×Ÿ ×œ× ×¨×¦×” (×œ× ×××•×¨ ×œ×§×¨×•×ª ×›×™ ×”×™× ×¨×¦×” ××”×”×ª×—×œ×”), ×”×ª×—×œ ××•×ª×”.
            if (!animationFrameId) {
                animate();
            }
        }
    });

     // ××™×¨×•×¢ ×œ×—×™×¦×” ×¢×œ ×›×¤×ª×•×¨ ×”×ª×—×œ×” ××—×“×©
     restartSimulationButton.addEventListener('click', () => {
         initializeParticles(); // ××ª×—×œ ×”×›×œ ××—×“×©
     });


    // ××™×¨×•×¢ ×œ×—×™×¦×” ×¢×œ ×›×¤×ª×•×¨ ×”×¦×’×ª/×”×¡×ª×¨×ª ×”×¡×‘×¨
    toggleExplanationButton.addEventListener('click', () => {
        if (explanationDiv.classList.contains('explanation-hidden')) {
            explanationDiv.classList.remove('explanation-hidden');
            // ×”×•×¡×¤×ª ×“×™×œ×™×™ ×§×¦×¨ ×›×“×™ ×©×”×× ×™××¦×™×” ×ª×™×¨××” ×˜×•×‘ ×™×•×ª×¨ ××—×¨×™ ×©×”-display ×”×©×ª× ×”
            setTimeout(() => {
                 explanationDiv.style.opacity = '1';
                 explanationDiv.style.transform = 'translateY(0)';
            }, 10); // ×“×™×œ×™×™ ×§×¦×¨ ×‘×™×•×ª×¨
            toggleExplanationButton.textContent = '×‘×˜×œ ×”×¡×‘×¨ ×¢×œ ×× ×˜×¨×•×¤×™×” ğŸ‘†'; // ×©×™× ×•×™ ×˜×§×¡×˜
            toggleExplanationButton.style.backgroundColor = '#dc3545'; // ×©×™× ×•×™ ×¦×‘×¢ ×œ×›×¤×ª×•×¨ '×‘×˜×œ'
        } else {
            explanationDiv.style.opacity = '0'; // ×”×ª×—×œ ×˜×©×˜×•×©
            explanationDiv.style.transform = 'translateY(10px)'; // ×”×ª×—×œ ×ª× ×•×¢×” ×œ××˜×”
            // ×”×¡×ª×¨ ×¤×™×–×™×ª ×œ××—×¨ ×¡×™×•× ×”×× ×™××¦×™×”
            explanationDiv.addEventListener('transitionend', function handleTransitionEnd() {

                 explanationDiv.removeEventListener('transitionend', handleTransitionEnd); // ×”×¡×¨ ××ª ×”×××–×™×Ÿ ×œ××—×¨ ×”×©×™××•×©
            }, { once: true }); // ×•×“× ×©×”×××–×™×Ÿ ×™×•×¤×¢×œ ×¤×¢× ××—×ª ×‘×œ×‘×“
            toggleExplanationButton.textContent = 'ğŸ¤” ××” ×‘×“×™×•×§ ×¨××™×ª×™? ×”×¦×’ ×”×¡×‘×¨ ×¢×œ ×× ×˜×¨×•×¤×™×” ğŸ¤”'; // ×©×™× ×•×™ ×˜×§×¡×˜ ×‘×—×–×¨×”
             toggleExplanationButton.style.backgroundColor = ''; // ×”×—×–×¨ ×¦×‘×¢ ×¨×’×™×œ

        }
    });

    // ××ª×—×•×œ ×¨××©×•× ×™ ×©×œ ×”×¡×™××•×œ×¦×™×”
    initializeParticles();

    // ×”×•×¡×¤×ª ×××–×™×Ÿ ×œ××™×¨×•×¢ ×©×™× ×•×™ ×’×•×“×œ ×—×œ×•×Ÿ ×›×“×™ ×œ×¢×“×›×Ÿ ××™××“×™×
    // window.addEventListener('resize', updateVesselDimensions); // ×–×” ×™×›×•×œ ×œ×”×™×•×ª ×™×§×¨ ××‘×—×™× ×ª ×‘×™×¦×•×¢×™×, × ×¢×“×›×Ÿ ×¨×§ ×‘-animate loop.


});
